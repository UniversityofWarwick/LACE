55 void LpcRun::run() {
56 
57  // Run the lpc algorithm over the required set of data.
58 
59  // Have we defined valid input & output pointers in initialise?
60  if (!theInput_) {
61  std::cerr<<"Input pointer is null. Please check that the input file "
62  <<"has a valid filename and format option..."<<std::endl;
63  return;
64  }
65 
66  if (!theOutput_) {
67  std::cerr<<"Output pointer is null. Please check that the output file "
68  <<"has a valid filename and format option..."<<std::endl;
69  return;
70  }
71 
72  // From the input, get the number of events & spatial dimensions
73  int nEvents = theInput_->getNEvents();
74  int nDim = theInput_->getNDimensions();
75 
76  std::cout<<"Number of events in "<<theInput_->getInputFileName()
77  <<" = "<<nEvents<<std::endl;
78  std::cout<<"Number of spatial dimensions for the hits = "<<nDim<<std::endl;
79 
80  // Loop over the events
81  // First, check to see if we have set (valid) indices of the
82  // first and last event to process
83  int firstIndex(0), lastIndex(nEvents);
84  if (theParameters_) {
85 
86  firstIndex = theParameters_->getFirstEvent();
87  lastIndex = theParameters_->getLastEvent() + 1;
88 
89  }
90 
91  // Check for negative values
92  if (firstIndex < 0) {firstIndex = 0;} // First event
93  if (lastIndex <= 0) {lastIndex = nEvents;} // All events
94 
95  // Loop over the events
96  for (int iEvt = firstIndex; iEvt < lastIndex; iEvt++) {
97 
98  // Get the event from the input
99  std::cout<<"Event "<<iEvt<<std::endl;
100  LpcEvent* theEvent = theInput_->getEvent(iEvt);
101 
102  if (theEvent != 0) {
103 
104  // Reconstruct this event to find curves, vertices & clusters
105  theProcess_->reconstruct(theEvent);
106 
107  // Store the output within the event
108  theOutput_->store(theEvent);
109 
110  } else {
111 
112  std::cerr<<"LpcRun: The event "<<iEvt<<" is null"<<std::endl;
113  }
114 
115  // Delete the event after we have stored/accessed its info
116  delete theEvent;
117 
118  } // Event loop
119 
120  // Finalise the input and output steps
121  theOutput_->finalise();
122  theInput_->finalise();
123 
124  std::cout<<"Results stored in "<<theOutput_->getOutputFileName()<<std::endl;
125 
126 }
