94 void LpcHitCollection::findCoordRange()
95 {
96 
97  for (int i = 0; i < nDim_; i++) {
98 
99  Eigen::VectorXd theColumn = Xi_.col(i);
100 
101  double r = theColumn.maxCoeff() - theColumn.minCoeff();
102 
103  // If the range is very small, or zero, then set the effective
104  // scaling factor to 1.0, i.e. no scaling is applied
105  if (fabs(r) < 1e-6) {r = 1.0;}
106 
107  theRange_(i) = r;

114 void LpcHitCollection::selectStartPoint()
115 {
116 
117  // Select the hit nearest to the weighted centroid as the starting point
118  LpcFunctions functions;
119  Eigen::VectorXd centroid = functions.getWeightedMean(Xi_, weights_);
120 
121  Eigen::VectorXd x0 = Eigen::VectorXd::Zero(nDim_);
122 
123  // Find the nearest hit to this centroid location
124  double minDistSq = 1e10;
125  for (int i = 0; i < Xi_.rows(); i++) {
126 
127  Eigen::VectorXd thePoint = Xi_.row(i);
128  double distSq = (thePoint - centroid).squaredNorm();
129  if (distSq < minDistSq) {
130  minDistSq = distSq;
131  x0 = thePoint;
132  }
133 
134  }
135 
136  Eigen::VectorXd scaledx0 = x0.cwiseQuotient(theRange_);
137 
138  startPoint_ = LpcPoint(-1, scaledx0, x0);
139 
140 }
