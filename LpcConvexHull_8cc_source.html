26 void LpcConvexHull::findLengths(LpcCluster* theCluster)
27 {
28 
29  // Find the convex hull lengths along the spatial dimensions of
30  // the cluster and store them in the cluster pointer. Originally,
31  // this method would require the calculation of the convex hull
32  // points with QHull. But, the extent of the hull is simply given
33  // by the size of the rectangular box that will enclose the
34  // cluster "ellipsoid", i.e. the lengths (max-min range) along
35  // each of the principal axes
36 
37  if (!theCluster) {return;}
38 
39  //Retrieve the matrix of cluster hit positions
40  Eigen::MatrixXd hitCoords = theCluster->getHitPositions();
41 
42  // We need to transform the point co-ordinates to lie along the principal axes.
43  // Find the eigenvectors of the covariance matrix of the hit co-ordinates
44  Eigen::MatrixXd covMatrix = functions_.getCovarianceMatrix(hitCoords);
45 
46  std::pair<Eigen::VectorXd, Eigen::MatrixXd> pcaEigen =
47  functions_.findNormEigenVectors(covMatrix);
48 
49  Eigen::MatrixXd eVectors = pcaEigen.second;
50 
51  Eigen::MatrixXd eV = eVectors.transpose();
52 
53  // Transform the hit co-ordinates. Row = point, col = x,y,z,...
54  Eigen::MatrixXd transCoords = hitCoords*eV;
55 
56  // Get the lengths along each axis
57  int nDim = hitCoords.cols();
58  Eigen::VectorXd lengths = Eigen::VectorXd::Zero(nDim);
59 
60  for (int i = 0; i < nDim; i++) {
61 
62  Eigen::VectorXd coordCol = transCoords.col(i);
63  double maxVal = coordCol.maxCoeff();
64  double minVal = coordCol.minCoeff();
65  double range = fabs(maxVal - minVal);
66 
67  lengths(i) = range;
68 
69  }
70 
71  theCluster->storeConvexHull(lengths);
72 
73 }
