45 void LpcFeatures::initialise()
46 {
47 
48  if (thePars_) {
49 
50  // The abs(cosAngle) distribution has spikes pointing downwards.
51  // We use 1.0 - abs(cosAngle) to get these peaks pointing upwards.
52  // Specify the minimum threshold cut for identifying peaks
53  cosAngleCut_ = thePars_->getCosAngleCut();
54 
55  // Specify what must be the next peaks minimum height w.r.t the
56  // previous peak, e.g. frac = 0.1 => 2nd peak must have
57  // height >= 10% of 1st peak, etc..
58  minPeakFrac_ = thePars_->getMinPeakFrac();
59 
60  // Set the distance (squared) limit for checking whether an lpc point
61  // corresponding to a peak is close to other lpc peak positions.
62  // In this case, the lpc point ranges are merged, avoiding duplicate
63  // vertices that will be essentially the same point
64  peakDiffSq_ = thePars_->getPeakDiffSq();
65 
66  // Select the peak finder method
67  peakFinder_ = thePars_->getPeakFinder();
68 
69  }

106 void LpcFeatures::findCosPeakRanges(LpcAbsCurve* theCurve)
107 {
108 
109  // First, get the peak positions for the 1-|cosAngle| distribution
110  allPeaks_.clear();
111  if (!theCurve) {return;}
112 
113  // Get the vector of LpcPoints for the curve
114  std::vector<LpcPoint> lpcPoints = theCurve->getLpcPoints();
115 
116  // Create a vector that will store the lpc number for each passed peak
117  std::vector<int> passedPeaks;
118 
119  // Also store the LpcPoints that correspond to a unique peak
120  std::vector<LpcPoint> peakCoord;
121 
122  // Need to also store a list of the peak bin ranges
123  std::vector<LpcBinRange> theRanges;
124 
125  // The number of lpc points, and other integers to check for bin boundaries
126  int nLpc = theCurve->getNLpcPoints();
127  int nLpc1 = nLpc - 1;
128  int nLpc2 = nLpc - 2;
129 
130  Eigen::VectorXd cosAngles = theCurve->getCosAngles();
131  Eigen::VectorXd absCosArray = Eigen::VectorXd::Zero(nLpc);
132  for (int i = 0; i < nLpc; i++) {
133 
134  absCosArray(i) = 1.0 - fabs(cosAngles(i));
135 
136  }
137 
138  // Get the vector of (bin, height) pairs for the cosine peaks
139  this->getCosPeaks(absCosArray);
140 
141  // If no peaks are found, stop
142  if (allPeaks_.size() < 1) {return;}
143 
144  // Sort the (bin, height) pairs, where height is in descending order
145  std::sort(allPeaks_.begin(), allPeaks_.end(), greater());
146 
147  int nPassedPeaks(0);
148  double prevPeakHeight(0.0);
149 
150  // Loop over the sorted peaks to find a first pass of the cosine ranges
151  std::vector<LpcPeakInfo>::const_iterator iter;
152  for (iter = allPeaks_.begin(); iter != allPeaks_.end(); ++iter) {
153 
154  LpcFeatures::LpcPeakInfo peak = *iter;
155  int peakBin0 = peak.bin_;
156  double peakHeight0 = peak.height_;
157 
158  // Sometimes the peak is +- 1 bin away.
159  // Check to see which has the largest height
160  int peakBin1 = peakBin0 - 1;
161  double peakHeight1(0.0);
162  if (peakBin1 >= 0) {
163  peakHeight1 = absCosArray(peakBin1);
164  }
165 
166  int peakBin2 = peakBin0 + 1;
167  double peakHeight2(0.0);
168  if (peakBin2 < nLpc) {
169  peakHeight2 = absCosArray(peakBin2);
170  }
171 
172  int peakBin(peakBin0);
173  double peakHeight(peakHeight0);
174  if (peakHeight1 > peakHeight) {
175  peakBin = peakBin1;
176  peakHeight = peakHeight1;
177  }
178  if (peakHeight2 > peakHeight) {
179  peakBin = peakBin2;
180  peakHeight = peakHeight2;
181  }
182 
183  // Check that this peak satisfies the minimum selection requirements.
184  // Exclude peaks that are right at the boundaries of the lpc curve, since
185  // we cannot look at lpc points on either side to find vertices etc..
186  if (peakBin > 0 && peakBin < nLpc1 && peakHeight > cosAngleCut_ &&
187  peakHeight > minPeakFrac_*prevPeakHeight) {
188 
189  int minBin = peakBin - dBin_;
190  int maxBin = peakBin + dBin_;
191  if (minBin < 1) {minBin = 1;}
192  if (maxBin > nLpc2) {maxBin = nLpc2;}
193 
194  LpcPoint lpcPoint = lpcPoints[peakBin];
195  Eigen::VectorXd lpcPos = lpcPoint.getCoords();
196 
197  // Check to see if the lpc point co-ordinate of this peak is
198  // very close to another one. If so, merge their ranges by
199  // taking the minimum and maximum limits of the lpc point integers
200 
201  bool gotSamePeak(false);
202  int samePeakInt(-1);
203 
204  int nPeakPoints = peakCoord.size();
205  for (int iP = 0; iP < nPeakPoints; iP++) {
206 
207  LpcPoint lpcPeak = peakCoord[iP];
208  Eigen::VectorXd peakPos = lpcPeak.getCoords();
209  Eigen::VectorXd diffPos = peakPos - lpcPos;
210  double distSq = diffPos.squaredNorm();
211 
212  if (distSq < peakDiffSq_) {
213 
214  // Peak points match
215  gotSamePeak = true;
216  samePeakInt = iP;
217 
218  LpcBinRange bin0 = theRanges[iP];
219  int minBin0 = bin0.getMinBin();
220  int maxBin0 = bin0.getMaxBin();
221  // Merge the minimum and maximum limits of the lpc point range
222  // by updating the minBin and maxBin values
223  if (minBin0 < minBin) {minBin = minBin0;}
224  if (maxBin0 > maxBin) {maxBin = maxBin0;}
225  // This peak matches a previous one. Stop this loop
226  break;
227 
228  }
229 
230  } // Loop over the current peak bins
231 
232  if (gotSamePeak) {
233 
234  // Set the range with the updated min/max lpc point limits
235  theRanges[samePeakInt] = LpcBinRange(minBin, maxBin);
236 
237  } else {
238  // We have a unique peak
239  nPassedPeaks += 1;
240  peakCoord.push_back(lpcPoint);
241 
242  LpcBinRange peakBinRange(minBin, maxBin);
243  theRanges.push_back(peakBinRange);
244 
245  passedPeaks.push_back(peakBin);
246 
247  }
248 
249  // Reset the previous peak height to this peak height
250  prevPeakHeight = peakHeight;
251 
252 
253  } // Checks on peak bin/height boundaries
254 
255  } // Loop over the sorted peak bins
256 
257  // Loop over the bin ranges and check to see if any ranges overlap.
258  // If they do, merge them. Also merge if any range < dBin_.
259  if (nPassedPeaks > 0 && theRanges.size() > 0) {
260 
261  // Sort the bin ranges in ascending (minimum bin) order
262  std::sort(theRanges.begin(), theRanges.end());
263 
264  // Define the vector of merged bin ranges
265  std::vector<LpcBinRange> mergedRanges;
266  // Add the first entry in the vector of sorted bin ranges
267  mergedRanges.push_back(theRanges[0]);
268 
269  for (int iR = 1; iR < nPassedPeaks; iR++) {
270 
271  LpcBinRange nextRange = theRanges[iR];
272  int lowNext = nextRange.getMinBin();
273  int highNext = nextRange.getMaxBin();
274 
275  int nMerge1 = mergedRanges.size() - 1;
276  LpcBinRange mergedBinR = mergedRanges[nMerge1];
277  int lowMerge = mergedBinR.getMinBin();
278  int highMerge = mergedBinR.getMaxBin();
279 
280  if (lowNext > lowMerge && highNext < highMerge) {
281  // The nextRange is already within the current merged range
282  continue;
283 
284  } else if (lowNext > lowMerge && lowNext < highMerge && highNext > highMerge) {
285  // Adjust the high end of the current merged range
286  mergedRanges[nMerge1].setMaxBin(highNext);
287 
288  } else if ((lowNext - highMerge) < dBin1_ && highNext > highMerge) {
289  // The next range is only dBin - 1 units away from the previous one.
290  // Adjust the high end of the current merged range
291  mergedRanges[nMerge1].setMaxBin(highNext);
292 
293  } else if (lowNext < lowMerge && highNext < highMerge && highNext > lowMerge) {
294  // Adjust the low end of the current merged range
295  mergedRanges[nMerge1].setMinBin(lowNext);
296 
297  } else if (lowNext < lowMerge && highNext > highMerge) {
298  // Adjust the low and high ends of the current merged range
299  mergedRanges[nMerge1] = nextRange;
300 
301  } else {
302  // Add new merged range interval
303  mergedRanges.push_back(nextRange);
304 
305  }
306 
307  } // Loop over passed peaks
308 
309  // Store the passed peaks and the final peak ranges to the lpc curve
310  theCurve->storeCosPeaks(passedPeaks, mergedRanges);
311 
312 
313  } // We have at least one passed peak
314 
315 }

346 void LpcFeatures::getSimplePeaks(const Eigen::VectorXd& absCosArray)
347 {
348 
349  // Find the peaks using a simple smoothing method by averaging
350  // neighbouring "bins" in the 1-|cosAngle| distribution. Peaks
351  // are identified if the 2 bins on either side of the smoothed
352  // distribution are lower in height
353 
354  allPeaks_.clear();
355 
356  int N = absCosArray.size();
357  if (N < 1) {return;}
358 
359  // Create a smoothed series of absCosValues
360  Eigen::VectorXd smoothArray = Eigen::VectorXd::Zero(N);
361  double maxVal(0.0);
362  int N1 = N - 1;
363  for (int iL = 1; iL < N1; iL++) {
364 
365  double v0 = absCosArray(iL-1);
366  double v1 = absCosArray(iL);
367  double v2 = absCosArray(iL+1);
368 
369  double smoothedVal = 0.5*(v0 + v2) + v1;
370  if (smoothedVal > maxVal) {maxVal = smoothedVal;}
371  smoothArray(iL) = smoothedVal;
372 
373  }
374 
375  // Find the smoothed peaks
376  double threshold = 0.05*maxVal;
377  int nS = smoothArray.size();
378 
379  std::vector<int> peakBins;
380 
381  for (int iS = 2; iS < nS-2; iS++) {
382 
383  double y0 = smoothArray(iS);
384  if (y0 < threshold) {continue;}
385 
386  double ym2 = smoothArray(iS-2);
387  double ym1 = smoothArray(iS-1);
388  double yp1 = smoothArray(iS+1);
389  double yp2 = smoothArray(iS+2);
390 
391  if (ym1 > ym2 && y0 > ym1 && yp1 < y0 && yp2 < yp1) {
392  // We have a peak
393  peakBins.push_back(iS+1);
394 
395  } else if (fabs(ym2) < threshold && fabs(yp2) < threshold) {
396 
397  // If the 2nd bins on either side are very small, we
398  // may have a narrow peak
399  if (y0 > ym1 && yp1 < y0) {
400  peakBins.push_back(iS+1);
401  }
402 
403  }
404 
405  }
406 
407  // Sometimes, the smoothing will move the peak by +-1 bin.
408  // Try to correct for this
409  int nPeaks = peakBins.size();
410 
411  std::vector<int> corrPeakBins;
412  for (int iP = 0; iP < nPeaks; iP++) {
413 
414  int pBin = peakBins[iP];
415 
416  int corrBin(pBin);
417  double yValue = absCosArray(pBin);
418  double thePeak(yValue);
419 
420  if (pBin > 1 && pBin < N1) {
421 
422  int pBin1 = pBin - 1;
423  int pBin2 = pBin + 1;
424  double yp0 = absCosArray(pBin1);
425  double yp2 = absCosArray(pBin2);
426 
427  if (yp0 > yValue) {
428  corrBin = pBin1;
429  thePeak = yp0;
430  } else if (yp2 > yValue) {
431  corrBin = pBin2;
432  thePeak = yp2;
433  }
434 
435  }
436 
437  // Store the lpc peak
438  LpcPeakInfo peakInfo;
439  peakInfo.bin_ = corrBin;
440  peakInfo.height_ = thePeak;
441  allPeaks_.push_back(peakInfo);
442 
443  }

448 void LpcFeatures::getTSpectrumPeaks(const Eigen::VectorXd& absCosArray)
449 {
450 
451  // Use the ROOT TSpectrum class for finding the peaks
452  // First, clear the vector of peak positions
453  allPeaks_.clear();
454 
455  if (absCosArray.size() < 1) {return;}
456 
457  // Check that the LPC_ROOT variable is set; otherwise the
458  // rest of the function is ignored
459 
460 #ifdef LPC_USE_ROOT
461 
462  TSpectrum spectrum(20, 10);
463 
464  float sigma(1.0);
465  double threshold(0.50);
466  bool bkgndRemove(false), markov(true);
467  int deconIter(5), avWindow(1);
468 
469  int nLpc = absCosArray.size();
470 
471  // Unfortunately, TSpectrum requires float "arrays"...
472  float* source = new float[nLpc];
473  float* destVector = new float[nLpc];
474  int i(0);
475  for (i = 0; i < nLpc; i++) {
476  source[i] = absCosArray(i);
477  destVector[i] = 0.0;
478  }
479 
480  int nPeaks = spectrum.SearchHighRes(source, destVector, nLpc, sigma, threshold,
481  bkgndRemove, deconIter, markov, avWindow);
482 
483  float* xPeaks = spectrum.GetPositionX();
484 
485  // Convert xPeaks to the integer values of the lpc point indices
486  for (i = 0; i < nPeaks; i++) {
487 
488  LpcPeakInfo peakInfo;
489  peakInfo.bin_ = int(xPeaks[i] + 0.5);
490  peakInfo.height_ = absCosArray(peakInfo.bin_);
491  allPeaks_.push_back(peakInfo);
492 
493  }
494 
495  // Delete the float arrays
496  delete [] source;
497  delete [] destVector;
498 
499 #endif
