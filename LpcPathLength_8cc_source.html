75 void LpcPathLength::initialise()
76 {
77  // Using the co-ordinate range and the scaled pathlengths,
78  // find the unscaled pathlengths and also calculate the
79  // delta pathlength values such that it starts at 0 and
80  // ends at the total pathlength.
81 
82  // First unscale the x,y,z components
83  LAxes_ = Eigen::MatrixXd::Zero(N_, nDim_);
84 
85  int i(0), j(0);
86  for (i = 0; i < N_; i++) {
87 
88  Eigen::VectorXd sLAxesRow = sLAxes_.row(i);
89  LAxes_.row(i) = sLAxesRow.cwiseProduct(theRange_);
90 
91  }
92 
93  // From these, find the unscaled 3d pathlengths
94  L_ = Eigen::VectorXd::Zero(N_);
95 
96  int N2 = N_/2;
97  int iterVal = N2 - 1;
98 
99  // Forward section
100  for (i = iterVal; i >= 0; i--) {
101 
102  j = i + 1;
103 
104  if (i == iterVal) {
105  L_(i) = 0.0;
106  } else {
107  Eigen::VectorXd diff = this->getDiff(i, j);
108  L_(i) = L_(j) + diff.norm();
109  }
110  }
111 
112  // Backward section
113  iterVal = N2;
114 
115  for (i = iterVal; i < N_; i++) {
116 
117  j = i - 1;
118 
119  Eigen::VectorXd diff = this->getDiff(i, j);
120  double length = diff.norm();
121 
122  if (i == iterVal) {
123  L_(i) = -length;
124  } else {
125  L_(i) = L_(j) - length;
126  }
127  }
128 
129  // Now find the ordered pathlengths, starting at 0
130  double max_sL = sL_.maxCoeff();
131  sdL_ = Eigen::VectorXd::Constant(N_, max_sL) - sL_;
132 
133  double max_L = L_.maxCoeff();
134  dL_ = Eigen::VectorXd::Constant(N_, max_L) - L_;
135 
136 }

143  diff = LAxes_.row(i) - LAxes_.row(j);
144  }
145 
146  return diff;
