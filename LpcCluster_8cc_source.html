169 void LpcCluster::update()
170 {
171 
172  if (!doUpdate_) {return;}
173 
174  // First, create an Eigen MatrixXd of the hit co-ordinates.
175  // We don't want to store this matrix as a private data member
176  // since it "repeats" the information already contained in the
177  // vector of LpcHit pointers. The matrix is only required so that
178  // we can use calculation methods in LpcFunctions.
179 
180  Eigen::MatrixXd hitPositions = this->getHitPositions();
181 
182  // Find the cluster centroid using the hit positions
183  this->calcCentroid(hitPositions);
184 
185  // Find the principal component axes (normalised eigenvectors) and
186  // their lengths (eigenvalues)
187  this->calcPCAxes(hitPositions);
188 
189  doUpdate_ = false;
190 
191 }

193 Eigen::MatrixXd LpcCluster::getHitPositions() const
194 {
195 
196  // Form an Eigen MatrixXd of the hit co-ordinates
197  // associated to the cluster
198 
199  // Number of hits in the cluster
200  int nHits = theHits_.size();
201 
202  if (nHits == 0) {
203  // Return an empty one row matrix
204  return Eigen::MatrixXd::Zero(1,nDim_);
205  }
206 
207  // Create and initialise the required hit position matrix
208  Eigen::MatrixXd hitPositions = Eigen::MatrixXd::Zero(nHits, nDim_);
209 
210  // Loop over the hits and store the position within each row
211  for (int iH = 0; iH < nHits; iH++) {
212 
213  LpcHit* theHit = theHits_[iH];
214  if (theHit) {
215 
216  hitPositions.row(iH) = theHit->getCoords();
217 
218  }
219 
220  }
221 
222  return hitPositions;
223 
224 }

253 void LpcCluster::calcPCAxes(const Eigen::MatrixXd& hitPositions)
254 {
255 
256  LpcFunctions functions;
257 
258  // Mean centred data
259  Eigen::MatrixXd meanData = functions.offsetPositions(hitPositions, centroid_);
260 
261  // Covariance matrix
262  Eigen::MatrixXd covMatrix = functions.formCovarianceMatrix(meanData);
263 
264  // Get the eigenvalues and normalised eigenvectors
265  std::pair<Eigen::VectorXd, Eigen::MatrixXd> eigenPair =
266  functions.findNormEigenVectors(covMatrix);
267 
268  // Set the internal data members
269  PCLengths_ = eigenPair.first;
270  PCAxes_ = eigenPair.second;
271 
272 }
