391  }

405 
406  TString clHitName("clHitX"); clHitName += i+1;

408  clTree_->Branch(clHitName.Data(), clHitCoords_[i]);
409  }
410 
411  // The hit weights

413  clTree_->Branch("clHitW", &clHitWeights_);
414 
415  }
416 
417  definedTrees_ = true;

421 void LpcRootOutput::storeCurve(const LpcCurve* mainCurve)
422 {
423 
424  // Store the main curve and all of its branches
425  if (!mainCurve) {return;}
426 
427  this->storeCurveDetails(mainCurve, false, 0);
428 
429  // Retrieve the branches
430  LpcBranchCollection* theBranches = mainCurve->getBranchCollection();
431 
432  if (theBranches) {
433 
434  // Get the number of generational levels for the branches
435  int nLevels = theBranches->getNumberLevels();
436 
437  // Loop over each branch generation
438  for (int i = 0; i < nLevels; i++) {
439 
440  int iG = i + 1; // Generation number
441  // Get the vector of branches
442  std::vector<LpcBranch*> branchVect = theBranches->getBranches(iG);
443 
444  // Loop over the branches
445  std::vector<LpcBranch*>::const_iterator iter;
446  for (iter = branchVect.begin(); iter != branchVect.end(); ++iter) {
447 
448  const LpcBranch* theBranch = *iter;
449  // Write out the information stored in the branch
450  if (theBranch) {
451 
452  //int branchId = theBranch->getIndex();
453  this->storeCurveDetails(theBranch, true, iG);
454 
455  } // Check if branch pointer exists
456 
457  } // Iteration over branches
458 
459  } // Branch generation loop
460 
461  } // Check for any stored branches
462 
463 }

465 void LpcRootOutput::storeCurveDetails(const LpcAbsCurve* theCurve, bool isABranch,
466  int branchGeneration)
467 {
468 
469  // Store either the main curve or one of its branches, following
470  // the methodology in LpcAbsCurve::print()
471 
472  if (!theCurve || !lpcTree_) {return;}
473 
474  int curveIndex = theCurve->getIndex();
475  int curveFlag = theCurve->getFlag();
476 
477  if (isABranch == false) {
478  // Main curve
479  curveId_ = curveIndex;
480  branchId_ = 0;
481  branchGen_ = 0;
482  } else {
483  // Branch
484  curveId_ = curveFlag;
485  branchId_ = curveIndex;
486  branchGen_ = branchGeneration;
487  }
488 
489  // The number of lpc points
490  nLpc_ = theCurve->getNLpcPoints();
491 
492  // Starting point
493  lpcStartPoint_.clear(); lpcStartPoint_.reserve(nDim_);
494 
495  LpcPoint startPoint = theCurve->getStartPoint();
496  Eigen::VectorXd x0 = startPoint.getCoords();
497  for (int i = 0; i < nDim_; i++) {
498  lpcStartPoint_.push_back(x0(i));
499  }
500 
501  // LpcPoints
502  std::vector<LpcPoint> lpcPoints = theCurve->getLpcPoints();
503 
504  // Eigenvectors
505  Eigen::MatrixXd eigenVectors = theCurve->getEigenVectors();
506 
507  for (int i = 0; i < nDim_; i++) {
508 
509  std::vector<double>* lpcPointVect = lpcPoints_[i];
510  lpcPointVect->clear(); lpcPointVect->reserve(nLpc_);
511 
512  std::vector<double>* lpcEigenVect = lpcEigenVectors_[i];
513  lpcEigenVect->clear(); lpcEigenVect->reserve(nLpc_);
514 
515  }
516 
517  // Cosine angles
518  Eigen::VectorXd cosAngles = theCurve->getCosAngles();
519  lpcCosAngles_->clear(); lpcCosAngles_->reserve(nLpc_);
520 
521  // Rho
522  Eigen::VectorXd rho = theCurve->getRho();
523  lpcRho_->clear(); lpcRho_->reserve(nLpc_);
524 
525  // c0
526  Eigen::VectorXd c0 = theCurve->getc0();
527  lpcC0_->clear(); lpcC0_->reserve(nLpc_);
528 
529  // Pathlengths object
530  LpcPathLength pathLength = theCurve->getPathLength();
531 
532  // Lambda
533  Eigen::VectorXd lambda = pathLength.getLambda();
534  lpcLambda_->clear(); lpcLambda_->reserve(nLpc_);
535 
536  // Lambda axes
537  Eigen::MatrixXd lambdaAxes = pathLength.getLambdaAxes();
538  for (int i = 0; i < nDim_; i++) {
539 
540  std::vector<double>* lpcLAxesVect = lpcLambdaAxes_[i];
541  lpcLAxesVect->clear(); lpcLAxesVect->reserve(nLpc_);
542 
543  }
544 
545  // Delta lambda
546  Eigen::VectorXd deltaLambda = pathLength.getDeltaLambda();
547  lpcPathLength_->clear(); lpcPathLength_->reserve(nLpc_);
548 
549  // Residuals object
550  LpcResiduals residuals = theCurve->getResiduals();
551 
552  // Lpc point residuals averaged over all nearest hits
553  Eigen::VectorXd lpcRes = residuals.getLpcResiduals();
554  lpcResiduals_->clear(); lpcResiduals_->reserve(nLpc_);
555 
556  // Weighted lpc point residuals averaged over all nearest hits
557  Eigen::VectorXd wLpcRes = residuals.getWeightedLpcResiduals();
558  wLpcResiduals_->clear(); wLpcResiduals_->reserve(nLpc_);
559 
560  // Fill the internal vectors with the above information
561 
562  for (int i = 0; i < nLpc_; i++) {
563 
564  lpcCosAngles_->push_back(cosAngles(i));
565  lpcRho_->push_back(rho(i));
566  lpcC0_->push_back(c0(i));
567  lpcLambda_->push_back(lambda(i));
568  lpcPathLength_->push_back(deltaLambda(i));
569  lpcResiduals_->push_back(lpcRes(i));
570  wLpcResiduals_->push_back(wLpcRes(i));
571 
572  Eigen::VectorXd lpcCoord = lpcPoints[i].getCoords();
573  Eigen::VectorXd mainEVector = eigenVectors.row(i);
574 
575  for (int j = 0; j < nDim_; j++) {
576 
577  lpcPoints_[j]->push_back(lpcCoord(j));
578  lpcEigenVectors_[j]->push_back(mainEVector(j));
579  lpcLambdaAxes_[j]->push_back(lambdaAxes(j));
580 
581  }
582 
583  }
584 
585  // HighRhoPoints
586  std::vector<LpcPoint> highRhoPoints = theCurve->getHighRhoPoints();
587  nHighRho_ = highRhoPoints.size();
588 
589  for (int i = 0; i < nDim_; i++) {
590 
591  std::vector<double>* highRhoVect = highRhoPoints_[i];
592  highRhoVect->clear(); highRhoVect->reserve(nHighRho_);
593 
594  }
595 
596  for (int i = 0; i < nHighRho_; i++) {
597 
598  Eigen::VectorXd highRhoP = highRhoPoints[i].getCoords();
599 
600  for (int j = 0; j < nDim_; j++) {
601  highRhoPoints_[j]->push_back(highRhoP(j));
602  }
603 
604  }
605 
606  // The number of cosine angle peaks
607  std::vector<int> cosPeakIndices = theCurve->getCosPeakIndices();
608  nCosPeaks_ = cosPeakIndices.size();
609  lpcCosPeaks_->clear(); lpcCosPeaks_->reserve(nCosPeaks_);
610 
611  for (int i = 0; i < nCosPeaks_; i++) {
612  lpcCosPeaks_->push_back(cosPeakIndices[i]);
613  }
614 
615  // Hit information
616  if (theHits_) {
617 
618  // The number of hits
619  nHits_ = theHits_->getNumberOfHits();
620 
621  // The hit co-ordinates
622  Eigen::MatrixXd hitCoords = theHits_->getCoords();
623 
624  for (int i = 0; i < nDim_; i++) {
625 
626  std::vector<double>* hitVect = hitCoords_[i];
627  hitVect->clear(); hitVect->reserve(nHits_);
628 
629  }
630 
631  // Hit-to-nearest-lpc point residuals
632  Eigen::VectorXd hitRes = residuals.getHitResiduals();
633  hitResiduals_->clear(); hitResiduals_->reserve(nHits_);
634 
635  // Weighted hit-to-nearest-lpc point residuals
636  Eigen::VectorXd wHitRes = residuals.getWeightedHitResiduals();
637  wHitResiduals_->clear(); wHitResiduals_->reserve(nHits_);
638 
639  // The index number of the nearest lpc point for each hit
640  Eigen::VectorXi hitNearLpc = residuals.getHitNearestLpc();
641  hitNearLpc_->clear(); hitNearLpc_->reserve(nHits_);
642 
643  for (int i = 0; i < nHits_; i++) {
644 
645  hitResiduals_->push_back(hitRes(i));
646  wHitResiduals_->push_back(wHitRes(i));
647  hitNearLpc_->push_back(hitNearLpc(i));
648 
649  Eigen::VectorXd hitPos = hitCoords.row(i);
650 
651  for (int j = 0; j < nDim_; j++) {
652 
653  hitCoords_[j]->push_back(hitPos(j));
654 
655  }
656 
657  }
658 
659  }
660 
661  // Store the data in the tree
662  lpcTree_->Fill();
663 
664 }

667 {
668 
669  if (!vtxTree_) {return;}
670 
671  nVtx_ = theVertices.size();
