24 std::pair<Eigen::VectorXd, Eigen::VectorXd>
25 LpcLineFitter::findLine(const Eigen::MatrixXd& data) const
26 {
27 
28  LpcFunctions functions;
29 
30  // Get the non-weighted mean of the data points
31  Eigen::VectorXd mean = functions.getMean(data);
32 
33  // Form a matrix of the centred data points
34  Eigen::MatrixXd meanData = functions.offsetPositions(data, mean);
35 
36  // Now find the Singular Value Decomposition (SVD) of the mean data
37  // which uses the covariance matrix of this adjusted data. The
38  // Rayleigh quotient is minimized and maximized by the eigen vectors of
39  // the covariance matrix that correspond to its smallest and largest
40  // eigenvalues, i.e. min eigen vector = normal vector of best-fit plane,
41  // max eigen vector = direction of best fit line. Note that the best line
42  // lies on the best plane; they are equivalent.
43 
44  // Get the (symmetric nDim*nDim) covariance matrix of the mean data
45  Eigen::MatrixXd covMatrix = functions.formCovarianceMatrix(meanData);
46 
47  // Create the Jacobi SVD matrix, with covMatrix = A = U S V*
48  Eigen::JacobiSVD<Eigen::MatrixXd> SVD(covMatrix, Eigen::ComputeThinU | Eigen::ComputeThinV);
49 
50  // The direction of the best fit line will be the eigenvector corresponding
51  // to the largest eigenvalue of the matrix U
52  Eigen::MatrixXd U = SVD.matrixU();
53 
54  // Get the first column of U, corresponding to the largest eigenvalue, since
55  // the Jacobi eigenvalues are automatically in descending order.
56 
57  int nDim = data.cols();
58  Eigen::VectorXd dirVect = Eigen::VectorXd::Zero(nDim);
59  // Ensure U is not empty
60  if (U.cols() > 0) {dirVect = U.col(0);}
61 
62  // Return the mean and unit direction vector along the regression line
63  return std::pair<Eigen::VectorXd, Eigen::VectorXd>(mean, dirVect);
64 
65 }
