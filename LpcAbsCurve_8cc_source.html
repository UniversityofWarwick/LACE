72 void LpcAbsCurve::findResiduals()
73 {
74 
75  int nLpc = lpcPoints_.size();
76  int nHits(0);
77  // Check if the pointer to the hit collection is null
78  if (!theHits_) {return;}
79 
80  nDim_ = theHits_->getNDimensions();
81  nHits = theHits_->getNumberOfHits();
82 
83  // Get the unscaled hit co-ordinates
84  const Eigen::MatrixXd hitArray = theHits_->getCoords();
85 
86  // Get the hit weights
87  const Eigen::VectorXd weights = theHits_->getWeights();
88 
89  Eigen::MatrixXd lpcArray = Eigen::MatrixXd::Zero(nLpc, nDim_);
90 
91  for (int i = 0; i < nLpc; i++) {
92  LpcPoint thePoint = lpcPoints_[i];
93  lpcArray.row(i) = thePoint.getCoords();
94  }
95 
96  // Arrays for the average residual of the closest data points and
97  // the number of closest points (for calculating the average recursively)
98  Eigen::VectorXd lpcResiduals = Eigen::VectorXd::Zero(nLpc);
99  Eigen::VectorXd lpcWeightRes = Eigen::VectorXd::Zero(nLpc);
100  // Store the number of points as floats, since we need to divide by them
101  Eigen::VectorXd lpcNClose = Eigen::VectorXd::Zero(nLpc);
102  Eigen::VectorXd lpcWClose = Eigen::VectorXd::Zero(nLpc);
103 
104  // Arrays for the nearest lpc residual for each data point
105  Eigen::VectorXd hitResiduals = Eigen::VectorXd::Zero(nHits);
106  Eigen::VectorXd hitWeightRes = Eigen::VectorXd::Zero(nHits);
107 
108  // Array storing the closest lpc point index number for each data point
109  Eigen::VectorXi hitNearestLpc = Eigen::VectorXi::Zero(nHits);
110 
111  // Here, we calculate two sets of residuals. The first is the average
112  // residual of the closest hit points for _each LPC point_.
113  // The next is the residual of the closest LPC point for _each hit point_
114 
115  // For the average residual per LPC point, we loop over all data hit points and find
116  // the LPC point that is closest to it. We then recursively add the residual
117  // distance to get the average residual for the matched LPC point.
118  // For the residual per hit point, just find the closest LPC point and store
119  // the separation distance
120 
121  // Loop over hits
122  for (int iH = 0; iH < nHits; iH++) {
123 
124  Eigen::VectorXd hitPoint = hitArray.row(iH);
125  double weight = weights(iH);
126  double minDistSq(0.0);
127 
128  int jClosest(0);
129 
130  // Loop over lpc points
131  for (int jL = 0; jL < nLpc; jL++) {
132 
133  Eigen::VectorXd lpcPoint = lpcArray.row(jL);
134  double distSq = (hitPoint - lpcPoint).squaredNorm();
135  if (jL == 0) {minDistSq = distSq;}
136 
137  if (distSq < minDistSq) {
138  minDistSq = distSq;
139  jClosest = jL;
140  }
141 
142  }
143 
144  // For the jClosest lpc point, find the average closest residual
145  // distance for each hit point using recursion
146  double N = lpcNClose(jClosest);
147  double N1 = N + 1.0;
148  double minDist = sqrt(minDistSq);
149 
150  lpcResiduals(jClosest) = (N*lpcResiduals(jClosest) + minDist)/N1;
151 
152  // Keep track of the weights for the jClosest lpc point
153  double w = lpcWClose(jClosest);
154  double w1 = w + weight;
155 
156  if (fabs(w1) > 1e-10) {
157  lpcWeightRes(jClosest) = (w*lpcWeightRes(jClosest) + minDist*weight)/w1;
158  }
159 
160  lpcNClose(jClosest) = N1;
161  lpcWClose(jClosest) = w1;
162 
163  hitResiduals(iH) = minDist;
164  hitWeightRes(iH) = minDist*weight;
165  hitNearestLpc(iH) = jClosest;
166 
167  }
168 
169  // Store the residuals
170  residuals_ = LpcResiduals(lpcResiduals, hitResiduals, lpcWeightRes,
171  hitWeightRes, hitNearestLpc);
172 
173 }
