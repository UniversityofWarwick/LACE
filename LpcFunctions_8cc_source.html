141 Eigen::MatrixXd LpcFunctions::formCovarianceMatrix(const Eigen::MatrixXd& meanData,
142  const Eigen::VectorXd& weights) const
143 {
144 
145  // Calculate the covariance matrix for the set of points
146  // within the (mean subtracted) data matrix (row = hit, col = x,y,z,...)
147  // with the vector of weights (each entry = charge or energy of hit)
148 
149  // Declare and initialise the covariance matrix.
150  // Get the number of co-ordinate variables
151  const int nCols = meanData.cols();
152  Eigen::MatrixXd covMatrix = Eigen::MatrixXd::Zero(nCols, nCols);
153 
154  // Loop over the set of data points (rows) and calculate the
155  // top part of the symmetric covariance matrix.
156  // The number of co-ordinate dimensions is generalised to nCols,
157  // but will usually be equal to 3 (x,y,z)
158 
159  for (int ip = 0; ip < meanData.rows(); ip++) {
160 
161  double w = weights(ip);
162 
163  // First, only fill the top right of the symmetric matrix
164  for (int ix = 0; ix < nCols; ix++) {
165 
166  double x_i = meanData(ip, ix);
167 
168  for (int iy = ix; iy < nCols; iy++) {
169 
170  double y_i = meanData(ip, iy);
171 
172  covMatrix(ix, iy) += w*x_i*y_i;
173 
174  }
175 
176  }
177 
178  }
179 
180 
181  // Fill the other part of the symmetric matrix
182  for (int ix = 0; ix < nCols; ix++) {
183 
184  for (int iy = ix; iy < nCols; iy++) {
185 
186  covMatrix(iy, ix) = covMatrix(ix, iy);
187 
188  }
189 
190  }
191 
192  // Now divide by the sum of the weights
193  double sum_weights = weights.sum();
194  covMatrix /= sum_weights;
195 
196  // Return the covariance matrix
197  return covMatrix;
198 
199 }

240 std::pair<Eigen::VectorXd, Eigen::MatrixXd>
241 LpcFunctions::findNormEigenVectors(const Eigen::MatrixXd& covMatrix) const
242 {
243 
244  // Covariance matrix will be symmetric
245  const int N = covMatrix.rows();
246 
247  Eigen::MatrixXd eigenVectors = Eigen::MatrixXd::Zero(N, N);
248  Eigen::VectorXd eigenValues = Eigen::VectorXd::Zero(N);
249 
250  // Eigenvectors and eigenvalues
251  Eigen::SelfAdjointEigenSolver<Eigen::MatrixXd> eigenSolver(covMatrix);
252 
253  // Check if the solver worked OK
254  if (eigenSolver.info() == Eigen::Success) {
255 
256  Eigen::MatrixXd origEigenVectors = eigenSolver.eigenvectors().real();
257  Eigen::VectorXd origEigenValues = eigenSolver.eigenvalues().real();
258 
259  // The eigenvectors are in ascending order. Reorder them to be descending.
260  // Also arrange the eigenvectors to be along each row
261  // of the returned eigenvector matrix
262 
263  for (int i = 0; i < N; i++) {
264 
265  int k = N - i - 1;
266  // Eigenvalues should be in descending order
267  eigenValues(i) = origEigenValues(k);
268 
269  for (int j = 0; j < N; j++) {
270  // Each eigenvector = matrix row
271  eigenVectors(i, j) = origEigenVectors(j, k);
272  }
273 
274  }
275 
276  }
277 
278  // The eigenvectors should already be normalised.
279  // Return the result as a STL pair
280 
281  return std::pair<Eigen::VectorXd, Eigen::MatrixXd>(eigenValues, eigenVectors);
282 
283 }

285 std::pair<double, double> LpcFunctions::getMeanAndRms(const Eigen::VectorXd& data) const
286 {
287 
288  // Calculate the mean and rms of the set of data recursively
289  double mean(0.0), rms(0.0);
290 
291  double N(0.0), N1(0.0);
292 
293  for (int i = 0; i < data.size(); i++) {
294 
295  // The current data value
296  double x = data(i);
297 
298  // Increment the number of data points
299  N += 1.0;
300  N1 = N - 1.0;
301 
302  double mean1 = mean; // Previous mean
303 
304  // Update the mean
305  if (fabs(N) > 1e-10) {
306  mean = (N1*mean1 + x)/N;
307  }
308 
309  // Update the (square) of the rms
310  double diff = mean - x;
311  if (fabs(N1) > 1e-10) {
312  rms += diff*diff*N/N1;
313  }
314 
315  }
316 
317  // Take the square root and divide by N
318  if (fabs(N) > 1e-10) {rms = sqrt(rms/N);}
319 
320  return std::pair<double, double>(mean, rms);
321 
322 }
