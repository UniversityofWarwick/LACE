50 LpcEvent* LpcTextInput::getEvent(int eventNo)
51 {
52 
53  // Check if the event number is valid
54  if (eventNo < 0 || eventNo >= nEvents_ || nDim_ < 1) {return 0;}
55 
56  // Usually, the events in the input file will be processed
57  // in sequential order by the LpcRun class
58 
59  // Retrieve the event number and number of hits, which should be
60  // on the present line if the input file is formatted correctly
61  std::pair<int, int> indices = this->getNextEventIndices();
62  int evtIndex = indices.first;
63  int nHits = indices.second;
64  int iEvt(0), iLine(0);
65 
66  if (eventNo < evtIndex) {
67 
68  // We have gone beyond the event. Reset the input streamer
69  // to the beginning of the file and try again.
70  this->initialise();
71 
72  // Get the first event index and number of events
73  indices = this->getNextEventIndices();
74  evtIndex = indices.first;
75  nHits = indices.second;
76 
77  }
78 
79  // Process the lines in the input file until we reach the correct event
80  LpcEvent* theEvent(0);
81 
82  for (iEvt = evtIndex; iEvt < nEvents_; iEvt++) {
83 
84  if (eventNo == iEvt) {
85 
86  // We have the required event
87 
88  // Create a new hit collection that will be stored
89  // and owned by the LpcEvent
90  LpcHitCollection* theHits = new LpcHitCollection(nDim_);
91 
92  // Loop over the hits and add them to the collection
93  // First, initialise the hitIndex_ to -1. This is always incremented
94  // before we create a new hit, and is assigned as the hit index value
95  hitIndex_ = -1;
96  for (iLine = 0; iLine < nHits; iLine++) {
97 
98  LpcHit* aHit = this->getNextHit();
99  // Add the hit if we have processed the data line OK.
100  // This will also increment hitIndex_ by one
101  if (aHit) {theHits->addHit(aHit);}
102 
103  }
104 
105  // Create the new LpcEvent
106  //std::cout<<"Creating new LpcEvent, evtNo = "<<eventNo<<", evtIndex = "
107  // <<evtIndex<<", nHits = "<<nHits<<std::endl;
108 
109  theEvent = new LpcEvent(eventNo, theHits);
110  break;
111 
112  } else if (eventNo > iEvt) {
113 
114  //std::cout<<"Skipping event "<<iEvt<<std::endl;
115 
116  // The event number we need has not yet been reached.
117  // Skip the lines for the current indexed event.
118  for (iLine = 0; iLine < nHits; iLine++) {
119 
120  // Skip the remaining lines until we reach the next event
121  getData_.ignore(maxNChar_, '\n');
122 
123  // Stop the loop if we have failed to read the line correctly
124  if (getData_.fail()) {break;}
125 
126  }
127 
128  // Get the next event index and number of events
129  indices = this->getNextEventIndices();
130  evtIndex = indices.first;
131  nHits = indices.second;
132 
133  } // The current event number is lower that what we need
134 
135  } // Loop over the number of possible events in the file
136 
137  return theEvent;
138 
139 }

162 LpcHit* LpcTextInput::getNextHit()
163 {
164  // Read the next line of hit data
165  std::string lineString("");
166  std::getline(getData_, lineString);
167 
168  // Stop if we have failed to read the line
169  if (getData_.fail()) {return 0;}
170 
171  // Split up the line according to white spaces
172  std::vector<std::string> lineVect = this->split(lineString, whiteSpace_);
173 
174  // Create a vector storing the hit information (co-ordinates and weight)
175  Eigen::VectorXd coords = Eigen::VectorXd::Zero(nDim_);
176 
177  // Convert the separated sub-strings into doubles
178  int nV = lineVect.size() - 1;
179  for (int i = 0; i < nV; i++) {
180 
181  double value = std::atof(lineVect[i].c_str());
182  coords(i) = value;
183 
184  }
185 
186  double weight = std::atof(lineVect[nV].c_str());
187 
188  // Increment the hit index
189  hitIndex_ += 1;
190  // Create the LpcHit pointer, then return it
191  LpcHit* theHit = new LpcHit(hitIndex_, coords, weight);
192 
193  return theHit;
194 
195 }

197 std::vector<std::string> LpcTextInput::split(const std::string& theString,
198  const std::string& splitter) const
199 {
200  // Code from STLplus
201  std::vector<std::string> result;
202 
203  if (!theString.empty() && !splitter.empty()) {
204 
205  for (std::string::size_type offset = 0;;) {
206 
207  std::string::size_type found = theString.find(splitter, offset);
208 
209  if (found != std::string::npos) {
210  std::string tmpString = theString.substr(offset, found-offset);
211  if (tmpString.size() > 0) {result.push_back(tmpString);}
212  offset = found + splitter.size();
213  } else {
214  std::string tmpString = theString.substr(offset, theString.size()-offset);
215  if (tmpString.size() > 0) {result.push_back(tmpString);}
216  break;
217  }
218  }
219  }
220 
221  return result;
