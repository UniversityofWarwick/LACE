86 {

89

91

93 
94  // Check for at least 1 dimension
95  if (nDim_ < 1) {return theBranches;}
96 
97  this->initialise();
98 
99  if (theHighRhoPoints.size() == 0 || branchLevel_ < 1) {return theBranches;}
100 
101  theHits_ = theHits;
102 
103  // Retrieve the co-ordinate and hit weight information.
104  // These will always be the same for all possible branches
105  Xi_ = theHits_->getScaledCoords();
106  weights_ = theHits_->getWeights();
107  N_ = Xi_.rows();
108  theRange_ = theHits_->getRange();
109  // Keep track of the index of the main curve that these branches will be associated with
110  mainCurveIndex_ = mainCurveIndex;
111 
112  // Select the starting point to initially be from the vector of highRhoPoints
113  startPoints_ = theHighRhoPoints;
114  // Start the index numbers for the branches at mainCurveIndex+1 (usually = 0+1 = 1)
115  int startIndex(mainCurveIndex+1);
116 
117  for (int iG = 0; iG < branchLevel_; iG++) {
118 
119  std::vector<LpcBranch*> branchVect = this->getGenBranches(startIndex);
120  // Add the branches, starting at generation level 1 (hence the +1)
121  theBranches->addBranches(iG+1, branchVect);
122 
123  // Update the starting index for the next generation
124  startIndex += branchVect.size();
125 
126  // Also update the new starting points to be the new high rho points
127  startPoints_.clear();
128  // Loop over the branches we have for this generation
129  std::vector<LpcBranch*>::const_iterator bIter;
130  for (bIter = branchVect.begin(); bIter != branchVect.end(); ++bIter) {
131 
132  const LpcBranch* theBranch = *bIter;
133  if (theBranch != 0) {

135  std::vector<LpcPoint> newRhoPoints = theBranch->getHighRhoPoints();
136 
137  std::vector<LpcPoint>::const_iterator pIter;
138  for (pIter = newRhoPoints.begin(); pIter != newRhoPoints.end(); ++pIter) {
139  // Add the high rho point to the starting points vector
140  startPoints_.push_back(*pIter);
141 
142  } // Loop over branch high rho points
143 
144  } // Branch is valid
145 
146  } // Loop over branches in current generation
147 
148  } // Generation loop

175 LpcBranch* LpcBranchAlgorithm::getBranch(int index)
176 {
177 
178  // First, we need to find the new forward position away from the curve
179  // We therefore need the 2nd largest eigenvector using the starting
180  // position given by the previous highRhoPoint ("x0_").
181 
182  // New forward position along the curve. Multiply the stepsize t by
183  // the branching gapsize (the total branch step) to move away
184  // from the initial curve
185  this->prepareNewStartPoint();
186 
187  Eigen::VectorXd new_x1 = mu_x_;
188  new_x1 += dx_;
189 
190  double kdexVal1 = functions_.kdex(Xi_, new_x1, h_);
191 
192  LpcCurve *curve1(0), *curve2(0);
193 
194  if (kdexVal1 > threshold_) {
195 
196  LpcPoint startPoint1(-1, new_x1);
197  startPoint1.unscale(theRange_);
198 
199  curve1 = lpcAlgorithm_.getCurve(index, theHits_, startPoint1,
200  jWeights_, E2Vector_,
201  LpcAlgorithm::Forward);
202 
203  }
204 
205  Eigen::VectorXd new_x2 = mu_x_;
206  new_x2 -= dx_;
207 
208  double kdexVal2 = functions_.kdex(Xi_, new_x2, h_);
209 
210  if (kdexVal2 > threshold_) {
211 
212  LpcPoint startPoint2(-1, new_x2);
213  startPoint2.unscale(theRange_);
214 
215  // The 2nd eigenvector needs to be "reversed" (minus sign)
216  curve2 = lpcAlgorithm_.getCurve(index, theHits_, startPoint2,
217  jWeights_, -E2Vector_,
218  LpcAlgorithm::Backward);
219 
220  }
221 
222  // From these curves, build the LpcBranch pointer.
223  int iter2 = thePars_->getNLpcPoints();
224  int iter = iter2/2;
225 
226  // Set the starting point for the branch result as the initial highRho point
227  LpcPoint startPoint(-1, x0_);
228  startPoint.unscale(theRange_);
229 
230  std::vector<LpcPoint> lpcPoints;
231  Eigen::MatrixXd eigenVectors = Eigen::MatrixXd::Zero(iter2, nDim_);
232  Eigen::VectorXd cosAngles = Eigen::VectorXd::Ones(iter2);
233  Eigen::VectorXd rho = Eigen::VectorXd::Zero(iter2);
234  Eigen::VectorXd c0 = Eigen::VectorXd::Ones(iter2);
235  std::vector<LpcPoint> highRhoPoints;
236  // For the path lengths
237  Eigen::VectorXd sLambda = Eigen::VectorXd::Zero(iter2);
238  Eigen::MatrixXd sLambdaAxes = Eigen::MatrixXd::Zero(iter2, nDim_);
239 
240  if (curve1) {
241 
242  std::vector<LpcPoint> lpcP1 = curve1->getLpcPoints();
243  Eigen::MatrixXd EVectors1 = curve1->getEigenVectors();
244  Eigen::VectorXd cAngles1 = curve1->getCosAngles();
245  Eigen::VectorXd rho1 = curve1->getRho();
246  Eigen::VectorXd c01 = curve1->getc0();
247 
248  LpcPathLength lpcPath1 = curve1->getPathLength();
249  Eigen::VectorXd sL1 = lpcPath1.getScaledLambda();
250  Eigen::MatrixXd sLAxes1 = lpcPath1.getScaledLambdaAxes();
251 
252  for (int i = 0; i < iter; i++) {
253 
254  lpcPoints.push_back(lpcP1[i]);
255  eigenVectors.row(i) = EVectors1.row(i);
256  cosAngles(i) = cAngles1(i);
257  rho(i) = rho1(i);
258  c0(i) = c01(i);
259 
260  sLambda(i) = sL1(i);
261  sLambdaAxes.row(i) = sLAxes1.row(i);
262 
263  }
264 
265  std::vector<LpcPoint> highRP1 = curve1->getHighRhoPoints();
266  std::vector<LpcPoint>::const_iterator iter;
267  for (iter = highRP1.begin(); iter != highRP1.end(); ++iter) {
268 
269  highRhoPoints.push_back(*iter);
270 
271  }
272 
273  }
274 
275  if (curve2) {
276

278  Eigen::MatrixXd EVectors2 = curve2->getEigenVectors();
279  Eigen::VectorXd cAngles2 = curve2->getCosAngles();
280  Eigen::VectorXd rho2 = curve2->getRho();
281  Eigen::VectorXd c02 = curve2->getc0();
282 
283  LpcPathLength lpcPath2 = curve2->getPathLength();

285  Eigen::MatrixXd sLAxes2 = lpcPath2.getScaledLambdaAxes();
286 
287  for (int i = iter; i < iter2; i++) {
288 
289  lpcPoints.push_back(lpcP2[i]);
290  eigenVectors.row(i) = EVectors2.row(i);
291  cosAngles(i) = cAngles2(i);
292  rho(i) = rho2(i);

306  }
307 
308  }
309 
310

312

325 void LpcBranchAlgorithm::prepareNewStartPoint()
326 {
327 
328  // Define the array of the kernel weights for the data hits
329  Eigen::VectorXd kernelWeights = Eigen::VectorXd::Zero(N_);
330 
331  // Loop over the data points & calculate the kernel weights
332  for (int ip = 0; ip < N_; ip++) {
333 
334  Eigen::VectorXd hitPoint = Xi_.row(ip);
335  double w = weights_(ip);
336 
337  kernelWeights(ip) = w*functions_.kernelFunction(hitPoint, x0_, h_);
338 
339  }
340 
341  // Find and set the weighted local mean
342  mu_x_ = functions_.getWeightedMean(Xi_, kernelWeights);
343 
344  // Subtract the local mean from the hit positions
345  Eigen::MatrixXd mean_sub = functions_.offsetPositions(Xi_, mu_x_);
346 
347  // Calculate the symmetric covariance matrix
348  Eigen::MatrixXd cov_x = functions_.formCovarianceMatrix(mean_sub, kernelWeights);
349 
350  // Get the normalised eigen vectors and values for the covariance matrix
351  std::pair<Eigen::VectorXd, Eigen::MatrixXd> eigenPair =
352  functions_.findNormEigenVectors(cov_x);
353 
354  Eigen::VectorXd eigenValues = eigenPair.first;
355  Eigen::MatrixXd eigenVectors = eigenPair.second;
356 
357  // Get the eigenvector with the 2nd (not 1st) largest eigenvalue
358  E2Vector_ = eigenVectors.row(1);
359 
360  // Set the required displacement for the branch starting point
361  dx_ = gapt_*E2Vector_;
362 
363  // Fill the jWeights array
364  jWeights_ = Eigen::VectorXd::Zero(N_);
365  for (int ik = 0; ik < N_; ik++) {
366 
367  Eigen::VectorXd hitPoint = Xi_.row(ik);
368  jWeights_(ik) = 1.0 - functions_.kernelFunction(hitPoint, mu_x_, h_);
369  jWeights_(ik) *= weights_(ik);
370 
371  }
