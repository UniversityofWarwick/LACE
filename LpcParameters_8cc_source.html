73 void LpcParameters::readParameterFile(const std::string& parameterFileName)
74 {
75 
76  // If the parameterFile is empty, use the default parameter values
77  if (parameterFileName.size() == 0) {
78  std::cout<<"Using the default parameter values."<<std::endl;
79  this->setDefaultValues();
80  return;
81  } else {
82  std::cout<<"Reading the lpc parameters from the file "
83  <<parameterFileName<<std::endl;
84  }
85 
86  std::ifstream getPar(parameterFileName.c_str());
87 
88  if (getPar.fail()) {
89  std::cerr<<"Could not open the parameter file "
90  <<parameterFileName<<"; using default values."<<std::endl;
91  this->setDefaultValues();
92  this->print();
93  return;
94  }
95 
96  std::string whiteSpace("\t");
97 
98  while (getPar.good()) {
99 
100  if (getPar.peek() == '\n') {
101 
102  // Finish reading the line
103  char c;
104  getPar.get(c);
105 
106  // Stop while loop if we have reached the end of the file
107  if (getPar.eof()) {break;}
108 
109  } else if (getPar.peek() == '#') {
110 
111  // Skip the comment line
112  getPar.ignore(1000, '\n');
113  getPar.putback('\n');
114 
115  // Stop while loop if we have reached the end of the file
116  if (getPar.eof()) {break;}
117 
118  } else {
119 
120  // Read the parameter line
121  std::string parName(""), parValue("");
122  getPar >> parName >> parValue;
123 
124  // Use lowercase for the parameter name
125  for (size_t i = 0; i < parName.size(); i++) {
126  parName[i] = std::tolower(parName[i], std::locale());
127  }
128 
129  // Find which parameter we have and set its value
130  if (parName == "infile") {
