49 void LpcAlgorithm::initialise()
50 {
51 
52  // Initialise the lpc parameters. This should be called by getCurve()
53  if (thePars_) {
54 
55  h_ = thePars_->getKernelWidth();
56  t_ = thePars_->getStepSize();
57  iter2_ = thePars_->getNLpcPoints();
58  iter_ = iter2_/2;
59  pen_ = thePars_->getAnglePenalisation();
60  rho0_ = thePars_->getEigenRatio();
61  boundary_ = thePars_->getBoundary();
62  convergence_ = thePars_->getConvergence();
63 
64  }
65 
66  // Initialise various matrices/vectors
67  // The local neighbourhood ("x" or "u") points with eigenvalue ratios > rho0
68  highRhoPoints_.clear();
69 
70  // The lpc points ("m(u)" or "mu"), equivalent to the local mean
71  lpcPoints_ = Eigen::MatrixXd::Zero(iter2_, nDim_);
72 
73  // The highest eigenvalue eigenvectors
74  eigenVectors_ = Eigen::MatrixXd::Zero(iter2_, nDim_);
75 
76  // The cosine between neighbouring eigenvectors
77  cosAngles_ = Eigen::VectorXd::Ones(iter2_);
78 
79  // The values of the cumulative path length
80  lambda_ = Eigen::VectorXd::Zero(iter2_);
81 
82  // The matrix of the cumulative path length along each axis (row = point, col = x,y,z..)
83  lambdaAxes_ = Eigen::MatrixXd::Zero(iter2_, nDim_);
84 
85  // The ratio of the eigenvalues between neighbouring eigenvectors
86  rho_ = Eigen::VectorXd::Zero(iter2_);
87 
88  // The c0 coefficients used to scale the kernel factor
89  c0_ = Eigen::VectorXd::Ones(iter2_);
90 
91 }

111 LpcCurve* LpcAlgorithm::getCurve(int curveIndex, const LpcHitCollection* theHits,
112  const LpcPoint& startPoint, const Eigen::VectorXd& weights,
113  const Eigen::VectorXd& lastEVect, int direction)
114 {
115 
116  if (!theHits) {return 0;}
117 
118  // Initialise parameters and internal arrays
119  nDim_ = theHits->getNDimensions();
120  // Check for at least 1 dimension
121  if (nDim_ < 1) {return 0;}
122 
123  this->initialise();
124 
125  // Retrieve the co-ordinate and hit weight information
126  Xi_ = theHits->getScaledCoords();
127  weights_ = weights;
128  x0_ = startPoint.getScaledCoords();
129 
130  startPoint_ = startPoint;
131  theRange_ = theHits->getRange();
132 
133  N_ = Xi_.rows();
134 
135  // Find the forward and backward curve sections, as required
136  if (direction == LpcAlgorithm::Forward) {
137 
138  this->forward(lastEVect);
139 
140  } else if (direction == LpcAlgorithm::Backward) {
141 
142  this->backward(lastEVect);
143 
144  } else {
145 
146  this->forward(lastEVect);
147  this->backward(lastEVect);
148 
149  }
150 
151  // Create a vector of the lpc points themselves
152  std::vector<LpcPoint> lpcPoints;
153 
154  for (int i = 0; i < iter2_; i++) {
155 
156  Eigen::VectorXd lp = lpcPoints_.row(i);
157 
158  LpcPoint thePoint(i, lp);
159  // Also find the unscaled co-ordinates
160  thePoint.unscale(theRange_);
161 
162  lpcPoints.push_back(thePoint);
163 
164  }
165 
166  // Store the scaled path lengths. This will also calculate the unscaled values
167  LpcPathLength lpcPaths(lambda_, lambdaAxes_, theRange_);
168 
169  // Construct the lpc curve pointer, storing all relevant info. This will
170  // also automatically calculate and store the lpc-to-hit residuals
171  int mainCurveFlag(-1);
172  LpcCurve* theCurve = new LpcCurve(curveIndex, startPoint_, lpcPoints, eigenVectors_,
173  cosAngles_, lpcPaths, rho_, c0_, highRhoPoints_,
174  theHits, mainCurveFlag);
175

181 void LpcAlgorithm::forward(const Eigen::VectorXd& lastEVect)
182 {
183 
184  int iterVal = iter_ - 1;
185 
186  // Set the initial local point
187  Eigen::VectorXd x = x0_;
188 
189  // Find the forward-going lpc points
190  for (int i = iterVal; i >= 0; i--) {
191 
192  // Finding lpc point i
193 
194  // Calculate the kernel weights for the lpc point
195  Eigen::VectorXd kernelWeights = Eigen::VectorXd::Zero(N_);
196 
197  // The kernel denominator factor c*h
198  double kernel_c = c0_(i)*h_;
199 
200  // Loop over the data points & calculate the kernel weights
201  for (int ip = 0; ip < N_; ip++) {
202 
203  Eigen::VectorXd hitPoint = Xi_.row(ip);
204  double w = weights_(ip);
205 
206  kernelWeights(ip) = w*functions_.kernelFunction(hitPoint, x, kernel_c);
207 
208  }
209 
210  // Find the weighted local mean
211  Eigen::VectorXd mu_x = functions_.getWeightedMean(Xi_, kernelWeights);
212 
213  // Subtract the local mean from the hit positions
214  Eigen::MatrixXd mean_sub = functions_.offsetPositions(Xi_, mu_x);
215 
216  // Calculate the symmetric covariance matrix
217  Eigen::MatrixXd cov_x = functions_.formCovarianceMatrix(mean_sub, kernelWeights);
218 
219  // Store the current lpc point
220  lpcPoints_.row(i) = mu_x;
221 
222  // Update the cumulative path lengths along the lpc
223  int j = i + 1;
224 
225  if (i == iterVal) {
226 
227  lambda_(i) = 0.0;
228  lambdaAxes_.row(i) = Eigen::VectorXd::Zero(nDim_);
229 
230  } else {
231 
232  Eigen::VectorXd prevLpcPoint = lpcPoints_.row(j);
233  Eigen::VectorXd lpcDiff = mu_x - prevLpcPoint;
234  // Get the distance (magnitude only)
235  for (int k = 0; k < lpcDiff.size(); k++) {
236  if (lpcDiff(k) < 0.0) {lpcDiff(k) *= -1.0;}
237  }
238 
239  lambda_(i) = lambda_(j) + lpcDiff.norm();
240  Eigen::VectorXd lAxes_j = lambdaAxes_.row(j);
241  lambdaAxes_.row(i) = lAxes_j + lpcDiff;
242 
243  }
244 
245 
246  // Get the normalised eigen vectors and values for the covariance matrix
247  std::pair<Eigen::VectorXd, Eigen::MatrixXd> eigenPair =
248  functions_.findNormEigenVectors(cov_x);
249 
250  Eigen::VectorXd eigenValues = eigenPair.first;
251  Eigen::MatrixXd eigenVectors = eigenPair.second;
252 
253  // Store the eigenvector with the largest eigenvalue
254  Eigen::VectorXd mainEVector = eigenVectors.row(0);
255  eigenVectors_.row(i) = mainEVector;
256 
257  // Store the ratio of the two largest eigenvalues
258  rho_(i) = 0.0;
259 
260  double maxEigenValue = eigenValues(0);
261  if (fabs(maxEigenValue) > 1e-10 && nDim_ > 1) {
262  rho_(i) = eigenValues(1)/maxEigenValue;
263  }
264 
265  if (i < iterVal && rho_(i) > rho0_ && rho_(j) < rho0_) {
266 
267  // Create a "high rho" LpcPoint object
268  LpcPoint theHighPoint(i, x);
269  // Obtain its unscaled co-ordinates
270  theHighPoint.unscale(theRange_);
271  // Store in the internal vector
272  highRhoPoints_.push_back(theHighPoint);
273 
274  }
275 
276  // Calculate the angle between successive eigenvectors. This needs
277  // the previous eigenvector
278  double lastEVectNorm(0.0);
279  if (i == iterVal) {lastEVectNorm = lastEVect.norm();}
280 
281  if (i == iterVal && lastEVectNorm > 0.0) {
282 
283  cosAngles_(i) = lastEVect.dot(eigenVectors_.row(i));
284 
285  } else if (i < iterVal) {
286 
287  cosAngles_(i) = eigenVectors_.row(j).dot(eigenVectors_.row(i));
288  }
289 
290  // Sign flipping
291  if (cosAngles_(i) < 0.0) {eigenVectors_.row(i) *= -1.0;}
292 
293  // Angle penalisation factor
294  if (pen_ > 0.0) {
295 
296  double a = std::pow(fabs(cosAngles_(i)), pen_);
297 
298  if (i == iterVal && lastEVectNorm > 0.0) {
299 
300  Eigen::VectorXd eigenRow_i = eigenVectors_.row(i);
301  eigenVectors_.row(i) = a*eigenRow_i + (1.0 - a)*lastEVect;
302 
303  } else if (i < iterVal) {
304 
305  Eigen::VectorXd eigenRow_i = eigenVectors_.row(i);
306  Eigen::VectorXd eigenRow_j = eigenVectors_.row(j);
307  eigenVectors_.row(i) = a*eigenRow_i + (1.0 - a)*eigenRow_j;
308 
309  }
310 
311  }
312 
313  // Step along the direction of the (penalised) eigenvector
314  x = eigenVectors_.row(i)*t_;
315  x += mu_x;
316 
317  // Check curve termination criteria
318  if (i > 0 && i < iterVal) {
319 
320  double conv_denom = fabs(lambda_(i) + lambda_(j));
321  double conv_ratio(0.0);
322  if (conv_denom > 1e-10) {
323  conv_ratio = fabs(lambda_(i) - lambda_(j))/conv_denom;
324  }
325 
326  if (conv_ratio < convergence_) {break;}
327 
328  // Adjust the boundary conditions for the next lpc point
329  // Since we are iterating downwards, this means the lpc point index
330  // number is decremented by 1
331  int k = i - 1;
332  if (conv_ratio < boundary_) {
333 
334  c0_(k) = 0.995*c0_(i);
335 
336  } else {
337 
338  c0_(k) = 1.01*c0_(i);
339  if (c0_(k) > 1.0) {c0_(k) = 1.0;}
340 
341  }
342 
343  }
344 
345  } // Loop over i
