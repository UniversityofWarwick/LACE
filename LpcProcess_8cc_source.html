67 void LpcProcess::reconstruct(LpcEvent* theEvent)
68 {
69 
70  if (!theEvent) {
71  std::cerr<<"Error in LpcProcess::reconstruct. The event is null"<<std::endl;
72  return;
73  }
74 
75  // Set-up the event so that the co-ordinates of the hit collection have been
76  // stored in Eigen matrices & scaled; this also finds the initial starting point
77  bool applyScaling(true);
78  if (doScaling_ == 0) {applyScaling = false;}
79 
80  theEvent->setUp(applyScaling);
81 
82  // Retrieve the hit collection from the event
83  LpcHitCollection* theHits = theEvent->getHitCollection();
84 
85  // Do the lpc calculation and obtain the main curve result
86  int mainCurveIndex(0);
87  LpcCurve* theCurve = theAlgorithm_->getCurve(mainCurveIndex, theHits, LpcAlgorithm::Both);
88 
89  if (theCurve != 0) {
90 
91  // Find any branches for this main curve
92  if (branchLevel_ > 0) {
93 
94  std::vector<LpcPoint> highRhoPoints = theCurve->getHighRhoPoints();
95 
96  LpcBranchCollection* theBranches =
97  theBranchAlgorithm_->getBranches(theHits, highRhoPoints, mainCurveIndex);
98 
99  theCurve->storeBranches(theBranches);
100 
101  }
102 
103 
104  // Find the feature points (peaks in 1-|cosAngle|) used for identifying
105  // vertices. This stores the peak ranges within the curve (and its branches)
106  theFeatures_->findFeatures(theCurve);
107 
108  // Store the curve in the event
109  theEvent->storeCurve(theCurve);
110 
111  // Find the vertices and obtain clusters for the main curve and any of its branches
112  if (clustering_ != 0) {
113 
114  LpcClusterData clusterData = theClusterAlgorithm_->findClusters(theCurve);
115 
116  // Store the cluster and vertexing information in the event. The event
117  // will then be responsible for the LpcCluster and LpcVertex pointers,
118  // i.e. they will be deleted when the event is deleted
119  std::vector<LpcVertex*> theVertices = clusterData.getVertices();
120  std::vector<LpcCluster*> theClusters = clusterData.getClusters();
121 
122  theEvent->storeVertices(theVertices);
123  theEvent->storeClusters(theClusters);
124 
125  // Determine which clusters could be showers
126  theShowerAlgorithm_->findShowers(theClusters);
127 
128  }
129 
130  }
